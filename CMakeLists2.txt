CMAKE_MINIMUM_REQUIRED(VERSION 3.0)

PROJECT(testing)

FIND_PACKAGE( OpenCV REQUIRED)

ADD_EXECUTABLE( ${PROJECT_NAME}
	src/main.cpp
	src/header.cpp
)

TARGET_INCLUDE_DIRECTORIES( ${PROJECT_NAME} PUBLIC
	include
)

TARGET_LINK_LIBRARIES( ${PROJECT_NAME} PUBLIC
	${OpenCV_LIBS}
)
